# TransitionGroup

> `TransitionGroup`可以很方便地进行列表过度动画

:::demo 基础用法

基础用法

```jsx
import { CssTransition, TransitionGroup, Button } from 'xl-vision'
export default () => {
  const countRef = React.useRef(0)

  const createElement = React.useCallback(() => {
    countRef.current++
    const key = countRef.current + ''
    return (
      <CssTransition key={key} classNames={'tg_item'}>
        <div className='tg_item' onClick={() => removeElement(key)}>
          第{countRef.current}个元素(点击删除)
        </div>
      </CssTransition>
    )
  }, [removeElement])

  const [elements, setElements] = React.useState(() => {
    const arr = []
    arr.push(createElement())
    return arr
  })

  const removeElement = React.useCallback(key => {
    setElements(prev => {
      return prev.filter(it => it.key !== key)
    })
  }, [])

  const addElement = React.useCallback(() => {
    setElements(prev => prev.concat(createElement()))
  }, [createElement])

  return (
    <div>
      <Button onClick={addElement}>ADD</Button>
      <div>
        <TransitionGroup>{elements}</TransitionGroup>
      </div> 
      <style jsx global>{`
        .tg_item {
          height: 3rem;
          color: #fff;
          line-height: 3rem;
          text-align: center;
          vertical-align: middle;
          background-color: #007bff;
          border-radius: 5px;
          margin: 5px 0;
          cursor: pointer;
        }

        .tg_item-enter-active,
        .tg_item-leave-active {
          transition: all 0.4s ease;
        }

        .tg_item-leave-to,
        .tg_item-enter {
          opacity: 0;
        }

        .tg_item-leave,
        .tg_item-enter-to {
          opacity: 1;
          transform: translateX(0);
        }

        .tg_item-enter {
          transform: translateX(-50px);
        }
        .tg_item-leave-to {
          transform: translateX(50px);
        }
      `}</style>
    </div>
  )
}
```

:::

## Transition API

| 参数 | 说明 | 类型 | 可选值 | 默认值 | 是否必填 |
| --- | --- | --- | --- | --- | --- |
| childFactory | 批量处理所有的子元素，通常用于批量使用Transition进行包裹 | (child: React.ReactElement) => React.ReactElement&#x3C;TransitionProps&#x3E; | - | (child: React.ReactElement) => child | false |
| children | 数组或单个元素，需要为Transition或其子类型，可以配合`childFactory`使用 | React.ReactElement[] | - | — | true |
